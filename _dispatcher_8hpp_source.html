180  {

183  auto& funcs = std::any_cast<std::vector<std::function<void(const Event&)>>&>(m_event_functions);
184 
185  // Discard unused branches and only compile used option.
186  if constexpr (action == meta::EventActions::TRIGGER)
187  {
188  // Exec each in parallel.

193  else if constexpr (action == meta::EventActions::ADD)

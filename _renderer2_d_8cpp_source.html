380  void Renderer2D::init(std::string_view layers)
381  {
382  const auto opt = json::parse_from_disk(layers);
383  if (opt != std::nullopt)
384  {
385  const auto& root = opt.value().at("layers");
386  for (const auto& [name, layer] : root.items())
387  {
388  m_layer_data.emplace(name, layer.get<int>());
389  }
390 
391  // Set pointers to layer data.
392  for (auto& [name, layer] : m_layer_data)
393  {
394  for (const auto& [key, atlas] : SL_HANDLE.texturebook()->get_all())
395  {
396  layer.m_batches.try_emplace(key);
397  layer.m_batches[key].add_texture(atlas.gl_texture());
398  }
399 
400  m_layers.push_back(&layer);
401  }
402 
403  // Make sure order is correct.
404  std::sort(std::execution::par, m_layers.begin(), m_layers.end(), [&](auto* left, auto* right) {
405  return left->get_layer() < right->get_layer();
406  });
407  }
408  else
409  {
410  GALAXY_LOG(GALAXY_FATAL, "Failed to create renderlayers from path: {0}.", layers);
411  }
412  }

460  void Renderer2D::submit(components::Primitive2D* data, components::Transform2D* transform)
461  {
462  // clang-format off
463  Renderable renderable = {
464  .m_vao = data->vao(),
465  .m_texture = 0,
466  .m_index_count = data->index_count(),
467  .m_configure_shader = [this, data, transform]()
468  {
469  this->m_point_shader.bind();
470  this->m_point_shader.set_uniform("u_colour", data->get_colour().normalized());
471  this->m_point_shader.set_uniform("u_transform", transform->get_transform());
472  }
473  };
474  // clang-format on
475 
476  switch (data->get_type())
477  {
478  case Primitives::CIRCLE:
479  case Primitives::ELLIPSE:
480  case Primitives::POLYGON:
481  case Primitives::POLYLINE:
482  renderable.m_type = GL_LINE_LOOP;
483  break;
484 
485  case Primitives::LINE:
486  renderable.m_type = GL_LINES;
487  break;
488 
489  case Primitives::POINT:
490  renderable.m_type = GL_POINTS;
491  break;
492  }
493 
494  m_layer_data.at(data->get_layer()).submit(renderable);
495  }
